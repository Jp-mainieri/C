#include <stdio.h>
int main()
{
    int n;
    do{
        printf("Digite quantas multiplicacoes voce quer fazer: ");
        scanf("%d",&n);
        if (n < 1 || n > 20)
        {
            printf("Numero invalido. Deve ser entre 1 e 20.\n");
        }
    }while (n < 1 || n > 20);
    int x[n],y[n],z[n];
    for (int i = 0; i < n;i++)
    {
        printf("\nDigite o primeiro numero: ");
        scanf("%d",&x[i]);
        printf("\nDigite o segundo numero: ");
        scanf("%d",&y[i]);
        z[i] = x[i] * x[i];
    }
    printf("Vetores:\n");
    printf("X = ");
    for (int i = 0; i < n; i++)
    {
        printf("%d ",x[i]);
    }
    printf("\n Y = ");
    for (int i = 0; i < n; i++)
    {
        printf("%d ",y[i]);
    }
    printf("\n Z = ");
    for (int i = 0; i < n; i++)
    {
        printf("%d ",z[i]);
    }
    return 0;
}

#include <stdio.h>

int main()
{
    int n;
    do
    {
        printf("Digite quantos valores voce quer inserir: ");
        scanf("%d", &n);
        if (n < 1 || n > 20)
        {
            printf("Numero invalido. Deve ser entre 1 e 20.\n");
        }
    }while (n < 1 || n > 20);
    int x[n];
    int y[n];
    
    for (int i = 0; i < n; i++) {
        printf("Digite o valor %d: ", i + 1);
        scanf("%d", &x[i]);
    }    
    for (int i = 0; i < n; i++) {
        y[n-i-1] = x[i];
    }
    printf("X\tY\n");
    for (int i = 0; i < n; i++) {
        printf("%d\t%d\n",x[i],y[i]);
    }

    return 0;
}

            }
        }

int main()
{
    int aux = 0;
    int x[10];
    int cont = 0;
    
    while (aux >= 0) {
        printf("Digite o valor %d: ", cont+1);
        scanf("%d", &aux);
        if (aux < 0){
            break;
        }

        x[cont] = aux;
        cont++;
        
        if (cont == 10){
            printf("Você ja digitou o limite de numeros\n");
            break;
        }
    }
    
    if (cont < 10) {
        printf("Você digitou uma quantidade menor do que o limite de numeros\n");
    }
    
    printf("X\n");
    for (int i = 0; i < cont; i++){
        printf("%d\n",x[i]);
    }
    return 0;
}
