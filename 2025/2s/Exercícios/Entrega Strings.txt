#include <stdio.h>

int main()
{
    char strVerificar[20], caractere;
    int count = 0;
    
    printf("Que caractere voce quer contar?");
    scanf("%c",&caractere);
    getchar();
    
    printf("\nDigite a string que voce quer verificar:");
    scanf("%[^\n]",strVerificar);
    
    for (int i = 0; i < sizeof(strVerificar); i++){
        if (strVerificar[i] == caractere){
            count++;
        }
    }
    printf("Na string: %s. Aparece o caractere: %c, %d vezes",strVerificar,caractere,count);
    
    return 0;
}

#include <stdio.h>
#include <string.h>

int main()
{
    char strOriginal[10], strInvertida[10];
    int len;

    printf("\nDigite a string que voce quer inverter:");
    scanf("%[^\n]",strOriginal);
    
    len = strlen(strOriginal);
    
    for (int i = len -1; i >= 0; i--){
        strInvertida[i] = strOriginal[len -1 -i];
    }
    strInvertida[len] = '\0';
    printf("String Original: %s\nString Invertida: %s",strOriginal,strInvertida);
    
    return 0;
}
