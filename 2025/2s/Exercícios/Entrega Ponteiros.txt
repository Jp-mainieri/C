#include <stdio.h>

int main()
{
	int var1,var2;
	int *pt1 = &var1;
	int *pt2 = &var2;
	int temp;

	printf("Insira dois valores:");
	scanf("%d %d", &var1, &var2);
	printf("%d %d\n",var1, var2);

	temp = *pt1;
	*pt1 = *pt2;
	*pt2 = temp;
	printf("%d %d",var1, var2);
	printf("%d %d",*pt1, *pt2);

	return 0;
}
