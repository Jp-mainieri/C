#include <stdio.h>

int main(){
    int A,B,C,D;
    A = 20;
    B = 0;
    C = 1;
    D = 10;
    if ((A - B + D) >= (C)){
        printf("True\n");
    }else{
        printf("False\n");
    }
    if ((A > (A + D)) || (C > B)){
        printf("True\n");
    }else{
        printf("False");
    }
    if (((A * D) > (C + A)) && (A > B)){
        printf("True\n");
    }else{
        printf("False\n");
    }
    if ((((A + B) / C) >= 0.9) && ((B * D) < 0)){
        printf("True\n");
    }else{
        printf("False\n");
    }
    if (((((A + C)/D) > 0.3) && (((B - C)<0) ) || ((C+A) > D))) {
        printf("True\n");
    }else{
        printf("False\n");
    }
    if ((((((A + C)/D) > 0.3) && ((B -C)<0) )) || ((C+A) > D)){
        printf("True\n");
    }else{
        printf("False\n");
    }
    return 0;
}

           "3 - Multiplicacao\n"

    {
    case 1:
        printf("Insira dois numeros reais:\n");
        scanf("%f %f",&n1,&n2);
        resultado = n1 + n2;
        break;
    case 2:
        printf("Insira dois numeros reais:\n");
        scanf("%f %f",&n1,&n2);
        resultado = n1 - n2;
        break;
    case 3:
        printf("Insira dois numeros reais:\n");
        scanf("%f %f",&n1,&n2);
        resultado = n1 * n2;
        break;
    case 4:
        printf("Insira dois numeros reais:\n");
        scanf("%f %f",&n1,&n2);

        break;
    default:
        printf("Opcao Invalida");
        break;
    }
    printf("Resultado = %.2f",resultado);

{

    int numero;
    int i = 2;
    printf("Digite um numero inteiro: ");
    scanf("%d",&numero);

    if (numero < 2) {
        ePrimo = true;
    } else {
        for (int i = 2; i < numero; i++) {
            if (numero % i == 0) {
                ePrimo = false;
                break;
            }
        }
    }

    if (ePrimo)
    {
        printf("O numero %d eh primo\n",numero);
    } else
        {
        printf("O numero %d nao eh primo\n",numero);
    }
    return 0;
}

#include <stdio.h>

int main()
{
    int num,count = 0,bignum = 0,smallnum;

    while (num != 0)
    {
        scanf("%d",&num);
        if (num > bignum)
        {
            bignum = num;
        }
        if (num < smallnum)
        {
            smallnum = num;
        }
        count++;
    }
    printf("A contagem eh %d\n",count);
    printf("O maior numero eh %d\n",bignum);
    printf("O menor numero eh %d\n",smallnum);

    return 0;
}

#include <stdio.h>

int main()
{
    int num,res = 1;

    do{
        res = 1;
        printf("Insert a number: ");
        scanf("%d",&num);
        for (int i = 1; i <= num; i++)
        {
            res = res * i;
        }
        printf("- %d factorial = %d\n",num,res);
    }while (num >= 0);

    return 0;
}
